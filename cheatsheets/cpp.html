Print Matrix

Print out a square matrix A with N elements. 
#include <iostream>
#include <iomanip>
for (int i=0; i < N; i++){
  for (int j=0; j < N; j++){
    cout << setw(8) << setprecision(4) << A[i][j];
  }
  cout << endl;
}
